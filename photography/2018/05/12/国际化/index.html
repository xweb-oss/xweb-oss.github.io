# 国际化 clever

# vue-i18n安装

npm install vue-i18n --save-dev

# 新建文件夹

新建一个lang 文件夹，在其下新建一个 zh.js 和 en.js文件 zh.js :

module.exports = {
 
　　　language: {
　　　　　　name: '中文'
　　　},
　　　navbar: {
　　　　　home: '首页',
　　}
}

en.js :

module.exports = {
 
　　　　language: {
　　　　　　name: 'English'
　　　　},
　　　　navbar: {
　　　　　　home: 'Home',
　　　　}
　　}

# 在main.js文件中引入

import VueI18n from 'vue-i18n';

Vue.use(VueI18n);

const i18n = new VueI18n({
　　　　locale: localStorage.getItem('language')||'zh', //使用localStorage缓存到本地，当下次使用时可默认当前使用语言
　　　　messages: {
 '　　　　　　zh': require('./common/lang/zh'),   // 根据自己的文件路径导入导入
　　　　　　'en': require('./common/lang/en')
　　　　}
　　})
new Vue({
　　el: '#app',
　　router,
　　i18n,
　　template: '<App/>',
　　components: { App }
})
